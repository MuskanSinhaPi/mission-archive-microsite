Using only Javascript
Embed to the body using only javascript
					
<script>
	if (!document.querySelector('#ft4begmek')) {
		var newDiv = document.createElement('div');
		newDiv.setAttribute('key', 'Instagram Feed  ');
		newDiv.setAttribute('class', 'ft');
		newDiv.setAttribute('id', 'ft4begmek');
		var newScript = document.createElement('script');
		newScript.setAttribute('src', 'https://wdg.fouita.com/widgets/0x3e57d1.js');
		newDiv.appendChild(newScript);

		document.body.appendChild(newDiv);
	}
</script>
